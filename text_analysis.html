clean_string = function(string){
  # Detect and replace '&#44'(comma) and `&#46`(period) from the string
  new_str = gsub("&#44", ",", string)
  new_str = gsub("&#46", ".", new_str)
  new_str = gsub("&#[0-9]+|&quot|\\(|\\)"," ", new_str)
  return(new_str)
}
